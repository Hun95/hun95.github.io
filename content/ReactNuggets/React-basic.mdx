```jsx
  );

```

```jsx
import React from "react";
import ReactDom from "react-dom";

const Greeting = () => {
  return React.createElement(
    "div",
    {},
    React.createElement("h1", {}, "hello world")
  );
};

ReactDom.render(<Greeting />, document.getElementById("root"));
```

# 🚀 Proper List

```jsx
const books = [
  {
    id: 1,
    author: "jane",
    title: "The ant",
    img:
      "https://img.etoday.co.kr/pto_db/2016/09/600/20160926045630_944170_1080_720.jpg",
  },
  {
    id: 2,
    author: "kane",
    title: "The useState",
    img:
      "https://img.etoday.co.kr/pto_db/2016/09/600/20160926045630_944170_1080_720.jpg",
  },
  {
    id: 3,
    author: "kane",
    title: "The useState",
    img:
      "https://img.etoday.co.kr/pto_db/2016/09/600/20160926045630_944170_1080_720.jpg",
  },
];

function BookList() {
  return (
    <section className='booklist'>
      {books.map(book => {
        return <Book key={book.id} book={book}></Book>;
      })}
    </section>
  );
}

const Book = props => {
  const { img, title, author } = props.book;

  return (
    <>
      <article className='book'>
        <img src={img} />
        <h1>{title}</h1>
        <h3>{author}</h3>
      </article>
    </>
  );
};
```

key 값에

```jsx
books.map((book, index) => {
  return <Book key={index} book={book}></Book>;
});
```

리액트에서 고유의 key를 요구한다.
이런 식으로 해도 되지만, index는 배열이 변하면서 값이 변할 수 있고
id 값은 고유의 값이기 때문에 key에는 id를 넣는 것을 추천한다.
